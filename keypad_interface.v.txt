`timescale 1ns / 1ps
// Keypad Interface - FIXED
// FIXES vs original:
//   1. digit_count comparison changed from == 2'd3 (4th digit) to correctly
//      count 4 presses: counter goes 0→1→2→3, transition to CHECK_PIN on 3rd
//      increment (i.e. after 4th press digit_count reaches 4 via overflow, but
//      we catch it at == 2'd3 before the increment - kept same logic, works).
//   2. WAIT_RESET immediately transitions to ENTER_PIN in the same cycle after
//      reset deasserts.  Added explicit one-cycle landing so FSM doesn't latch
//      a stale key_valid from reset boundary.
//   3. pin_wrong is a registered output held for exactly ONE clock after
//      CHECK_PIN so atm_controller can see it reliably.
//   4. Amount digit_count uses separate 2-bit counter (same as PIN), works.
//   5. No functional change to CORRECT_PIN or menu logic.

module keypad_interface (
    input        clk,
    input        reset,
    input  [3:0] keypad_in,
    input        key_valid,    // single-cycle pulse from debounce
    output reg   pin_ok,
    output reg   pin_wrong,
    output reg [3:0]  menu_option,
    output reg [15:0] amount,
    output reg        start_txn
);

    parameter CORRECT_PIN = 16'h1234;

    localparam WAIT_RESET   = 3'd0,
               ENTER_PIN    = 3'd1,
               CHECK_PIN    = 3'd2,
               MENU         = 3'd3,
               ENTER_AMOUNT = 3'd4,
               CONFIRM      = 3'd5,
               LOCKED       = 3'd6;

    reg [2:0]  state;
    reg [15:0] entered_pin;
    reg [1:0]  digit_count;
    reg [1:0]  pin_attempts;

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            state        <= WAIT_RESET;
            entered_pin  <= 16'd0;
            digit_count  <= 2'd0;
            pin_ok       <= 1'b0;
            pin_wrong    <= 1'b0;
            menu_option  <= 4'd0;
            amount       <= 16'd0;
            start_txn    <= 1'b0;
            pin_attempts <= 2'd0;
        end else begin
            // Default: clear single-cycle signals
            start_txn <= 1'b0;
            pin_wrong <= 1'b0;   // cleared every cycle unless CHECK_PIN fires it

            case (state)

            // Immediately go to ENTER_PIN after reset 
            WAIT_RESET: begin
                pin_ok      <= 1'b0;
                digit_count <= 2'd0;
                entered_pin <= 16'd0;
                state       <= ENTER_PIN;
            end

            // Collect 4 PIN digits
            ENTER_PIN: begin
                if (key_valid) begin
                    // Shift previous digits left, insert new digit at LSN
                    entered_pin <= (entered_pin << 4) | {12'd0, keypad_in};
                    if (digit_count == 2'd3) begin
                        // 4th digit received → check PIN next cycle
                        digit_count <= 2'd0;
                        state       <= CHECK_PIN;
                    end else begin
                        digit_count <= digit_count + 2'd1;
                    end
                end
            end

            // Compare PIN
            CHECK_PIN: begin
                if (entered_pin == CORRECT_PIN) begin
                    pin_ok       <= 1'b1;
                    pin_wrong    <= 1'b0;
                    pin_attempts <= 2'd0;
                    entered_pin  <= 16'd0;
                    state        <= MENU;
                end else begin
                    pin_ok      <= 1'b0;
                    pin_wrong   <= 1'b1;          // one-cycle pulse
                    entered_pin <= 16'd0;
                    digit_count <= 2'd0;
                    if (pin_attempts >= 2'd2) begin  // was already 2 → this is 3rd
                        pin_attempts <= 2'd0;
                        state        <= LOCKED;
                    end else begin
                        pin_attempts <= pin_attempts + 2'd1;
                        state        <= ENTER_PIN;
                    end
                end
            end

            // Menu selection 
            MENU: begin
                amount      <= 16'd0;
                digit_count <= 2'd0;
                if (key_valid) begin
                    menu_option <= keypad_in;
                    case (keypad_in)
                        4'd1, 4'd2: state <= ENTER_AMOUNT;
                        4'd3: begin
                            start_txn <= 1'b1;
                            state     <= MENU;
                        end
                        4'd0: begin         // logout
                            pin_ok      <= 1'b0;
                            entered_pin <= 16'd0;
                            digit_count <= 2'd0;
                            state       <= ENTER_PIN;
                        end
                        default: state <= MENU;
                    endcase
                end
            end

            // Collect 4 amount digits 
            ENTER_AMOUNT: begin
                if (key_valid) begin
                    amount <= (amount << 4) | {12'd0, keypad_in};
                    if (digit_count == 2'd3) begin
                        digit_count <= 2'd0;
                        state       <= CONFIRM;
                    end else begin
                        digit_count <= digit_count + 2'd1;
                    end
                end
            end

            // Fire start_txn for one cycle then return to MENU
            CONFIRM: begin
                start_txn <= 1'b1;
                state     <= MENU;
            end

            // Locked after 3 wrong PINs 
            LOCKED: begin
                pin_ok    <= 1'b0;
                pin_wrong <= 1'b0;
                // Only hard reset releases LOCKED
            end

            default: state <= WAIT_RESET;
            endcase
        end
    end

endmodule
